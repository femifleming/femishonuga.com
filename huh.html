<script>
    // Function to generate a random color in hexadecimal format
    function getRandomColor() {
        var letters = '0123456789ABCDEF';
        var color = '#';
        for (var i = 0; i < 6; i++) {
            color += letters[Math.floor(Math.random() * 16)];
        }
        return color;
    }

    // Function to change the color of every letter on the page
    function changeLetterColors() {
        var allNodes = document.createTreeWalker(
            document.body,
            NodeFilter.SHOW_TEXT,
            null,
            false
        );
        var currentNode;

        // Iterate through all text nodes in the document
        while (currentNode = allNodes.nextNode()) {
            var text = currentNode.nodeValue.trim();
            var newText = '';

            // Change the color of each letter
            for (var i = 0; i < text.length; i++) {
                newText += '<span style="color: ' + getRandomColor() + ';">' + text[i] + '</span>';
            }

            // Replace the text node with colored spans
            var parent = currentNode.parentNode;
            var temp = document.createElement('div');
            temp.innerHTML = newText;
            while (temp.firstChild) {
                parent.insertBefore(temp.firstChild, currentNode);
            }
            parent.removeChild(currentNode);
        }
    }

    // Change letter colors every 20ms
    setInterval(changeLetterColors, 20);
</script>
